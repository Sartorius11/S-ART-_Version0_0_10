create TABLE users (
  idUsuario INT PRIMARY KEY,
  nombre VARCHAR(50) NOT NULL,
  email VARCHAR(100) NOT NULL,
  salt VARCHAR(50) NOT NULL,
  pass VARCHAR(100) NOT NULL,
  pais VARCHAR(50)  NULL,
  artista_fav VARCHAR(50) Null,
);
